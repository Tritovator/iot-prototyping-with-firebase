JavaScript

board.on("ready", function () {
  var button = new five.Button(8);

  button.on("down", function () { ref.set("true"); });

  button.on("up", function () { ref.set("false"); });

  var led = five.Led(13);
  ref.on("value", function(snap) {
    if(snap.val() == "true") {
      led.on();
      setTimeout(function() { led.off(); }, 500);
    } 
  });
});

Video Playing Code

// Connect to a demo Firebase
var ref =  new Firebase("https://firebutton.firebaseio-demo.com");
      
ref.child("button").on("value", function (snapshot) {
  // Read the value
  var buttonValue = snapshot.val();
  // If it's true, play the video
  if (buttonValue) {
    $('#player').tubeplayer({
      initialVideo: "dQw4w9WgXcQ"
    });
    $('#player').tubeplayer("play");
  }
});
